    int maxScore(vector<int>& v, int k) {
        
        int sum=0;
        int n=v.size();
        for(int i=0;i<k;i++){
            sum+=v[i];
        }
        int res=sum;
        int itr=1;
        for(int i=k-1;i>=0;i--){
            sum-=v[i];
            sum+=v[n-itr];
            res=max(res,sum);
            itr++;
        }
        return res;

    }
