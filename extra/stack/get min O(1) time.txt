#include <bits/stdc++.h>

pair<string,string> process_str(string str){
    
    int left=0;
    pair<string,string> res;
    for(int i=0;i<str.size();i++){
        if(str[i]==' '){
            res.first=str.substr(left,i-left);
            left=i+1;
            break;
        }
    }
    
    if(res.first==""){
        res.first=str.substr(left);
    }else{
        res.second=str.substr(left);
    }
    return res;
    
}
