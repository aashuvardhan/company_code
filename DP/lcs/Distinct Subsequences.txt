/*
https://leetcode.com/problems/distinct-subsequences/description/

Given two strings s and t, return the number of distinct subsequences of s which equals t.

The test cases are generated so that the answer fits on a 32-bit signed integer.

 

Example 1:

Input: s = "rabbbit", t = "rabbit"
Output: 3
Explanation:
As shown below, there are 3 ways you can generate "rabbit" from s.
rabbbit
rabbbit
rabbbit
Example 2:

Input: s = "babgbag", t = "bag"
Output: 5
Explanation:
As shown below, there are 5 ways you can generate "bag" from s.
babgbag
babgbag
babgbag
babgbag
babgbag
 

Constraints:

1 <= s.length, t.length <= 1000
s and t consist of English letters.

*/


int numDistinct(string s, string t) {

        int m=s.size();
        int n=t.size();

        vector<vector<unsigned int>> tab(m+1,vector<unsigned int>(n+1,0));
        //tab[i][j] stores the number of ways to form string t (up to index j) using string s (up to index i).
        for(int i=0;i<=n;i++){
            tab[0][i]=0;
        }
        for(int i=0;i<=m;i++){
            tab[i][0]=1;
        }

        for(int i=1;i<=m;i++){
            for(int j=1;j<=n;j++){
                if(s[i-1]==t[j-1]){
                    // Match found: Add ways by USING the character + ways by IGNORING it
                    tab[i][j]=tab[i-1][j-1] + tab[i-1][j];
                }else{
                    // No match: We must IGNORE the current character in s
                    tab[i][j]=tab[i-1][j];
                }
            }
        }

        return tab[m][n];
        
    }
