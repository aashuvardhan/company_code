/*


Given the root of a binary tree, return the length of the longest path, where each node in the path has the same value. This path may or may not pass through the root.

The length of the path between two nodes is represented by the number of edges between them.

 

Example 1:


Input: root = [5,4,5,1,1,null,5]
Output: 2
Explanation: The shown image shows that the longest path of the same value (i.e. 5).
Example 2:


Input: root = [1,4,5,4,4,null,5]
Output: 2
Explanation: The shown image shows that the longest path of the same value (i.e. 4).
 

Constraints:

The number of nodes in the tree is in the range [0, 104].
-1000 <= Node.val <= 1000
The depth of the tree will not exceed 1000.

*/


pair<int,int> dfs(TreeNode* root,int& res){
        if(root==NULL){
            return {-5000,0};
        }
        if(root->left==NULL and root->right==NULL){
            return {root->val,0};
        }
        pair<int,int> p1=dfs(root->left,res);
        pair<int,int> p2=dfs(root->right,res);

        if(root->val==p1.first and root->val==p2.first){
            res=max(res,p1.second+p2.second+1+1);
            return {root->val,max(p1.second+1,p2.second+1)};
        }else if(root->val==p1.first and root->val!=p2.first){
            res=max(res,p1.second+1);
            return {root->val,p1.second+1};
        }else if(root->val!=p1.first and root->val==p2.first){
            res=max(res,p2.second+1);
            return {root->val,p2.second+1};
        }else{
            res=max(res,0);
            return {root->val,0};
        }
    }

    int longestUnivaluePath(TreeNode* root) {

        
        int res=0;
        dfs(root,res);


        return res;

    }
