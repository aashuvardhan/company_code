#include <bits/stdc++.h>
using namespace std;


struct ListNode{
    int val;
    ListNode* next;
    ListNode(int x){
        val=x;
        next=NULL;
    }

};



void add_one(ListNode* node,int& carry){
    if(node==NULL){
        return;
    }
    add_one(node->next,carry);
    if(node->val==9){
        carry=1;
        node->val=0;
    }else{
        node->val+=carry;
        carry=0;
    }
}

void traversal(ListNode* head){
    
    while(head!=NULL){
        cout<<head->val<<"  ";
        head=head->next;
    }
    cout<<endl;
}


int main() {
	// your code goes here
    ListNode* head=new ListNode(9);
    head->next=new ListNode(9);
    head->next->next=new ListNode(9);
    head->next->next->next=new ListNode(9);
    
    traversal(head);
    
    int carry=0;
    add_one(head,carry);
    
    if(carry==1){
        ListNode* temp=new ListNode(1);
        temp->next=head;
        head=temp;
    }
    traversal(head);

}
