#include <bits/stdc++.h>
using namespace std;


struct Trienode{
    bool isEnd;
    int nodecount;
    Trienode* children[26];
    
    Trienode(){
        isEnd=false;
        nodecount=0;
        for(int i=0;i<26;i++){
            children[i]=NULL;
        }
    }
};

void insert(Trienode* root,string str){
    Trienode* curr=root;
    
    for(int i=0;i<str.size();i++){
        int index=str[i]-'a';
        if(curr->children[index]==NULL){
            curr->children[index]=new Trienode();
            
        }
        curr->nodecount++;
        curr=curr->children[index];
    }
    curr->isEnd=true;
}

int countWordsEqualTo(Trienode* root,string str){
    Trienode* curr=root;
    int res=0;
    for(int i=0;i<str.size();i++){
        int index=str[i]-'a';
        if(curr->children[index]==NULL){
            return 0;
        }
        res=curr->nodecount;
        curr=curr->children[index];
        
    }
    if(curr->isEnd==true){
        return res;
    }else{
        return 0;
    }
}


int countWordsStartingWith(Trienode* root,string str){
    Trienode* curr=root;
    int res=0;
    for(int i=0;i<str.size();i++){
        int index=str[i]-'a';
        if(curr->children[index]==NULL){
            return 0;
        }
        res=curr->nodecount;
        curr=curr->children[index];
    }
    return res;
}

bool search(Trienode* root,string str){
    Trienode* curr=root;
    
    for(int i=0;i<str.size();i++){
        int index=str[i]-'a';
        
        if(curr->children[index]==NULL){
            return false;
        }
        curr=curr->children[index];
    }
    return curr->isEnd;
}


void erase(Trienode* root,string str){
    
    
    if(search(root,str)==false){
        return;
    }
    
    Trienode* curr=root;
    
    for(int i=0;i<str.size();i++){
        int index=str[i]-'a';
        curr->nodecount--;
        curr=curr->children[index];
    }
}


int main() {
	// your code goes here
	
	Trienode* root=new Trienode();
	
	insert(root,"apple");
	insert(root,"apple");
	insert(root,"apple");
	insert(root,"apple");
	cout<<countWordsEqualTo(root,"apple")<<endl;
	
	insert(root,"app");
	cout<<countWordsStartingWith(root,"app")<<endl;
	cout<<countWordsStartingWith(root,"apple")<<endl;
	
	
	erase(root,"apple");
	cout<<countWordsStartingWith(root,"app")<<endl;
	cout<<countWordsStartingWith(root,"apple")<<endl;

}
